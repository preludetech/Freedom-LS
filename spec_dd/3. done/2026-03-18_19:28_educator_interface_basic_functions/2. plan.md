# Implementation Plan: Educator Interface Basic Functions

This plan implements four interconnected features using TDD. Each task follows RED-GREEN-REFACTOR.

**Skills to use throughout:**
- `testing` skill for all test writing (TDD)
- `template` skill for all template creation/editing
- `frontend-styling` skill for Tailwind classes — always check `tailwind.components.css` before using raw Tailwind, and extract repeated patterns into component classes
- `alpine-js` skill for client-side interactivity (modals, tabs) — always add `x-transition` when using `x-show`
- `multi-tenant` skill when working with SiteAwareModel
- `icon-usage` skill when adding icons to buttons
- `brand-guidelines` skill for visual design decisions

**Design note:** The spec mentions a `TabbedInstanceView` subclass. This plan instead adds optional `tabs` support directly to `InstanceView` (with conditional rendering). This avoids an unnecessary class hierarchy while achieving the same result — any InstanceView can define tabs without requiring a different base class.

```python
class CreateInstanceAction(FormPanelAction):
    """Base class for actions that create a new instance via a modal form.

    Subclasses must define: form_class, form_title, label, action_name.
    Subclasses must implement: get_success_url(instance) and get_created_event_name().
    """
    variant: str = "primary"
    submit_buttons = [
        {"label": "Save and add another", "variant": "outline", "name": "action", "value": "save_and_add"},
        {"label": "Save", "variant": "primary"},
    ]

    def get_success_url(self, instance: Model) -> str:
        """Return the URL to redirect to after successful creation."""
        raise NotImplementedError

    def get_created_event_name(self) -> str:
        """Return the HTMX event name to trigger on 'save and add another' (e.g. 'cohortCreated')."""
        raise NotImplementedError

    def _render_empty_form(self, request: HttpRequest) -> str:
        """Re-render the modal form with an empty/unbound form (for 'save and add another')."""
        form = self.form_class()
        return render_to_string(
            "panel_framework/partials/modal_form.html",
            {"form": form, "form_title": self.form_title, "submit_buttons": self.submit_buttons},
            request=request,
        )

    def has_permission(self, request: HttpRequest, instance: Model | None = None) -> bool:
        model = self.form_class.Meta.model
        app_label = model._meta.app_label
        model_name = model._meta.model_name
        return request.user.has_perm(f"{app_label}.add_{model_name}")

    def form_valid(self, request: HttpRequest, form: forms.ModelForm) -> HttpResponse:
        instance = form.save()
        if request.POST.get("action") == "save_and_add":
            html = self._render_empty_form(request)
            response = HttpResponse(html)
            response["HX-Trigger"] = self.get_created_event_name()
            return response
        response = HttpResponse(status=204)
        response["HX-Redirect"] = self.get_success_url(instance)
        return response
```

**Implementation in `panel_framework/tabs.py`:**
```python

@dataclass
class Tab:
    label: str
    panels: dict[str, type[Panel]]
```

Each `Tab` owns its panels directly as a `dict[str, type[Panel]]`. The InstanceView no longer needs a separate top-level `panels` dict when using tabs — all panels are defined within their tabs.

**Update `InstanceView` in `panel_framework/views.py`:**
```python
class InstanceView:
    panels: dict[str, type[Panel]] = {}
    tabs: dict[str, Tab] | None = None

    def get_tab(self, tab_name: str) -> Tab:
        """Get a tab by name."""
        return self.tabs[tab_name]

    def get_panel_from_tab(self, tab_name: str, panel_name: str) -> type[Panel]:
        """Look up a panel class within a specific tab."""
        tab = self.get_tab(tab_name)
        return tab.panels[panel_name]

    def render(self, request, base_url="", active_tab: str | None = None):
        if self.tabs:
            return self._render_tabbed(request, base_url, active_tab=active_tab)
        return self._render_flat(request, base_url)

    def _render_flat(self, request, base_url):
        # existing render logic using self.panels
        ...

    def _render_tabbed(self, request, base_url, active_tab: str | None = None):
        # If active_tab is provided, render that tab's panels inline;
        # otherwise default to the first tab.
        # Pass tab URLs and active_tab name to the template so Alpine.js
        # can initialise with the correct tab and update the browser URL.
        ...

    def render_tab(self, request, tab_name: str, base_url: str) -> str:
        """Render panels for a specific tab (used by lazy-load HTMX calls)."""
        tab = self.tabs[tab_name]
        tab_url = f"{base_url.rstrip('/')}/__tabs/{tab_name}"
        rendered_panels = []
        for name, panel_class in tab.panels.items():
            panel_url = f"{tab_url}/__panels/{name}"
            panel = panel_class(instance=self.instance)
            rendered_panels.append(panel.render(request, base_url=panel_url, panel_name=name))
        return render_to_string(
            "panel_framework/partials/tab_panels.html",
            {"rendered_panels": rendered_panels},
            request=request,
        )
```

**URL structure for tabs:**
- `cohorts/{pk}/__tabs/{tab_name}` — lazy-load all panels for a tab
- `cohorts/{pk}/__tabs/{tab_name}/__panels/{panel_name}` — reload a single panel within a tab
- `cohorts/{pk}/__tabs/{tab_name}/__panels/{panel_name}/__actions/{action}` — action on a panel within a tab

Panel URLs are always scoped under their tab. This means each panel's `base_url` includes its tab path, so HTMX reload URLs (`hx-get` on panel containers) naturally route through the correct tab. The `_resolve_path` handler walks the segments: resolve instance → resolve tab by name → resolve panel by name within that tab's `panels` dict.

### Task 3.2: Tab container template

Use the `template` skill, `alpine-js` skill, and `frontend-styling` skill.

Create `panel_framework/templates/panel_framework/partials/tab_container.html`:

Uses Alpine.js for tab state. First tab renders immediately; other tabs lazy-load via HTMX on first click and stay in the DOM once loaded.

**Important:** Check `tailwind.components.css` for existing tab-related component classes before writing raw Tailwind. If no tab component classes exist, add reusable classes for tab bar, active tab, and inactive tab states to `tailwind.components.css` rather than using inline utility classes. Run `npm run tailwind_build` after adding new component classes.

The template receives:
- `tabs` — a list of tab context dicts (preserving insertion order from the `tabs` dict), each with `name`, `label`, `load_url`, `rendered_panels` (only for the active tab — whichever tab the server chose to render inline).
- `active_tab` — the name of the tab whose panels are rendered inline (server-set, defaults to first tab).
- `base_url` — the instance's base URL (e.g. `/educator/cohorts/5`), used for constructing tab URLs for `history.pushState`.

**URL behaviour:**
- When a tab is clicked, the browser URL updates to `{base_url}/__tabs/{tab_name}` via `history.pushState`. This ensures the URL is shareable and bookmarkable.
- When the page loads, the server determines which tab is active from the URL and passes `active_tab` to the template, so Alpine.js initialises with the correct tab.
- The first tab (default) uses the plain instance URL (e.g. `/educator/cohorts/5`) rather than `/__tabs/first_tab_name` to keep the default URL clean.

```html
<div x-data="{
        activeTab: '{{ active_tab }}',
        loadedTabs: { '{{ active_tab }}': true },
        baseUrl: '{{ base_url }}',
        defaultTab: '{{ tabs.0.name }}',
        switchTab(name) {
            this.activeTab = name;
            if (!this.loadedTabs[name]) {
                this.loadedTabs[name] = true;
                this.$nextTick(() => htmx.trigger(
                    document.getElementById('tab-content-' + name),
                    'load-tab'
                ));
            }
        }
     }"
     @popstate.window="
        const match = window.location.pathname.match(/__tabs\/([^/]+)/);
        switchTab(match ? match[1] : defaultTab);
     ">
    {# Tab bar #}
    <div role="tablist" class="flex border-b border-border mb-6">
        {% for tab in tabs %}
            <button @click="switchTab('{{ tab.name }}');
                           history.pushState({}, '',
                               '{{ tab.name }}' === defaultTab
                                   ? baseUrl
                                   : baseUrl + '/__tabs/{{ tab.name }}'
                           );"
                    :class="activeTab === '{{ tab.name }}'
                        ? 'border-b-2 border-primary text-primary font-semibold'
                        : 'text-muted hover:text-foreground'"
                    class="px-4 py-2 -mb-px transition-colors"
                    role="tab"
                    :aria-selected="activeTab === '{{ tab.name }}'">
                {{ tab.label }}
            </button>
        {% endfor %}
    </div>

    {# Tab panels #}
    {% for tab in tabs %}
        <div x-show="activeTab === '{{ tab.name }}'"
             x-transition
             x-cloak
             id="tab-content-{{ tab.name }}"
             role="tabpanel">
            {% if tab.name == active_tab %}
                {{ tab.rendered_panels|safe }}
            {% else %}
                <div hx-get="{{ tab.load_url }}"
                     hx-trigger="load-tab once"
                     hx-swap="innerHTML">
                    <c-loading-indicator />
                </div>
            {% endif %}
        </div>
    {% endfor %}
</div>
```

### Task 3.3: Wire up `__tabs` URL handling

Update `_resolve_path` in `educator_interface/views.py` to handle `__tabs/{tab_name}` and nested segments:

- `__tabs/{tab_name}/__panels/{panel_name}` — resolve the tab by name, look up panel by name in `tab.panels`, render that single panel
- `__tabs/{tab_name}/__panels/{panel_name}/__actions/{action}` — resolve panel within tab, then resolve action on that panel

**Direct tab URL access (full-page rendering):**

When `__tabs/{tab_name}` is requested:
- **HTMX request** (`HX-Request` header present): return just the rendered tab panels as a fragment (for lazy-loading).
- **Non-HTMX request** (direct browser navigation): render the full instance page with `active_tab=tab_name`, so the user sees the complete page with the correct tab selected. This is done by calling `instance_view.render(request, base_url, active_tab=tab_name)` and wrapping it in the full page template, same as a normal instance view render.

This ensures tab URLs are directly shareable and bookmarkable — visiting `cohorts/5/__tabs/details` in the browser renders the full cohort page with the "Details" tab active.

This keeps panel and action resolution tab-scoped so that URLs are directly addressable.

**Implementation in `panel_framework/actions.py`:**
```python
class DeleteAction(PanelAction):
    label = "Delete"
    variant = "danger"
    action_name = "delete"

    def __init__(self, success_url: str = ""):
        self.success_url = success_url

    def get_cascade_summary(self, instance: Model) -> list[str]:
        """Use Django's Collector to show what will be cascade-deleted."""
        from django.db.models.deletion import Collector
        collector = Collector(using=instance._state.db)
        collector.collect([instance])
        summary = []
        for model, objs in collector.data.items():
            if model is not type(instance):
                count = len(objs)
                if count:
                    summary.append(f"{count} {model._meta.verbose_name_plural}")
        return summary

    def render(self, request, instance, base_url):
        """Render danger button + confirmation modal.

        Note: DeleteAction is an instance-level action (not panel-level),
        so it receives `instance` directly rather than a `panel`.
        """
        cascade = self.get_cascade_summary(instance)
        return render_to_string(
            "panel_framework/partials/delete_confirmation.html",
            {"instance": instance, "cascade_summary": cascade,
             "delete_url": f"{base_url}/__actions/delete", "variant": self.variant},
            request=request,
        )

    def handle_submit(self, request, instance):
        instance.delete()
        response = HttpResponse(status=204)
        response["HX-Redirect"] = self.success_url
        return response

    def has_permission(self, request, instance=None):
        if instance is None:
            return False
        model_name = instance._meta.model_name
        app_label = instance._meta.app_label
        return request.user.has_perm(f"{app_label}.delete_{model_name}", instance)
```
