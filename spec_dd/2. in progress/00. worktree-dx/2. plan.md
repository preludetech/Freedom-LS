# pseudocode
import re
from pathlib import Path

_cached_branch: str | None = None
_cache_set: bool = False


def get_current_branch(base_dir: Path | None = None) -> str | None:
    global _cached_branch, _cache_set
    if _cache_set:
        return _cached_branch

    if base_dir is None:
        from django.conf import settings
        base_dir = settings.BASE_DIR

    # ... existing file-based .git parsing logic from context_processors.py ...
    result = _detect_branch(base_dir)

    _cached_branch = result
    _cache_set = True
    return result


def branch_to_db_name(branch: str) -> str:
    sanitized = branch.lower()
    sanitized = re.sub(r"[^a-z0-9]", "_", sanitized)
    # Truncate sanitized branch to 50 chars, then prefix with db_
    return f"db_{sanitized[:50]}"
```

**Caching strategy:** Do NOT use `@functools.lru_cache` because the function has a `base_dir=None` default that resolves lazily — `lru_cache` would cache the `None` key without resolving it. Instead, use a simple module-level cache with a `_cache_set` flag. Provide a `_clear_branch_cache()` function for tests.

**Truncation:** The spec says truncate branch name to 50 chars. `branch_to_db_name` truncates the *sanitized branch part* to 50 chars before adding the `db_` prefix. This gives a max db name of 53 chars (`db_` + 50), and max test db name of 58 chars (`test_db_` + 50), well within PostgreSQL's 63-char limit.

# pseudocode
def get_cached_site(request):
    if not hasattr(request, "_cached_site"):
        force_name = getattr(settings, "FORCE_SITE_NAME", None)
        if force_name:
            try:
                request._cached_site = Site.objects.get(name=force_name)
            except Site.DoesNotExist:
                request._cached_site = get_current_site(request)
        else:
            request._cached_site = get_current_site(request)
    return request._cached_site
```

## Task 4: `dev_db_init.sh` script

### 4.1 Create `dev_db_init.sh` at project root

```bash
#!/bin/sh
# Creates per-branch dev and test databases in the shared PostgreSQL container.
# Idempotent: skips creation if database already exists.
# Assumes the docker container from dev_db/ is already running on port 6543.

set -e

# Detect branch and sanitize (same logic as branch_to_db_name in git_utils.py)
BRANCH=$(git branch --show-current 2>/dev/null || true)
if [ -z "$BRANCH" ]; then
    DB_NAME="db"
else
    DB_NAME="db_$(echo "$BRANCH" | tr '[:upper:]' '[:lower:]' | sed 's/[^a-z0-9]/_/g' | cut -c1-50)"
fi
TEST_DB_NAME="test_${DB_NAME}"

PSQL="PGPASSWORD=password psql -h 127.0.0.1 -p 6543 -U pguser -d postgres"

# Create dev database if not exists
eval $PSQL -tc "SELECT 1 FROM pg_database WHERE datname = '${DB_NAME}'" | grep -q 1 || \
    eval $PSQL -c "CREATE DATABASE ${DB_NAME};"
eval $PSQL -c "GRANT ALL PRIVILEGES ON DATABASE ${DB_NAME} TO pguser;"

# Create test database if not exists
eval $PSQL -tc "SELECT 1 FROM pg_database WHERE datname = '${TEST_DB_NAME}'" | grep -q 1 || \
    eval $PSQL -c "CREATE DATABASE ${TEST_DB_NAME};"
eval $PSQL -c "GRANT ALL PRIVILEGES ON DATABASE ${TEST_DB_NAME} TO pguser;"

echo "Databases ready: ${DB_NAME}, ${TEST_DB_NAME}"
```

Make executable: `chmod +x dev_db_init.sh`

## Task 5: `dev_db_delete.sh` script

### 5.1 Create `dev_db_delete.sh` at project root

```bash
#!/bin/sh
# Drops per-branch dev and test databases from the shared PostgreSQL container.
# Terminates active connections before dropping.

set -e

# Detect branch and sanitize (same logic as dev_db_init.sh)
BRANCH=$(git branch --show-current 2>/dev/null || true)
if [ -z "$BRANCH" ]; then
    echo "Not on a branch. Nothing to delete."
    exit 0
fi
DB_NAME="db_$(echo "$BRANCH" | tr '[:upper:]' '[:lower:]' | sed 's/[^a-z0-9]/_/g' | cut -c1-50)"
TEST_DB_NAME="test_${DB_NAME}"

PSQL="PGPASSWORD=password psql -h 127.0.0.1 -p 6543 -U pguser -d postgres"

# Terminate connections and drop dev database
eval $PSQL -c "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname = '${DB_NAME}' AND pid <> pg_backend_pid();" 2>/dev/null || true
eval $PSQL -c "DROP DATABASE IF EXISTS ${DB_NAME};"

# Terminate connections and drop test database
eval $PSQL -c "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname = '${TEST_DB_NAME}' AND pid <> pg_backend_pid();" 2>/dev/null || true
eval $PSQL -c "DROP DATABASE IF EXISTS ${TEST_DB_NAME};"

echo "Dropped: ${DB_NAME}, ${TEST_DB_NAME}"
```

Make executable: `chmod +x dev_db_delete.sh`
