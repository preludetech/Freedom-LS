### `course_icon.py` — pseudocode

```python
# freedom_ls/student_interface/course_icon.py
from django.conf import settings
from django.utils.html import escape, format_html
from django.utils.safestring import SafeString

from freedom_ls.icons.backend import get_icon_backend
from freedom_ls.icons.loader import load_iconify_data
from freedom_ls.icons.mappings import ICON_SETS
from freedom_ls.icons.semantic_names import SEMANTIC_ICON_NAMES


class IconResolutionError(RuntimeError):
    """Raised when a literal-glyph render expects a suffixed glyph that isn't present."""


_FALLBACK_RE = ...  # ^[a-z0-9_-]+:[a-z0-9_-]+$


def _active_set_name() -> str:
    return getattr(settings, "FREEDOM_LS_ICON_SET", "heroicons")


def _render_literal(set_name: str, glyph: str, variant: str, css_class: str, aria_label: str) -> SafeString | None:
    """Try to render <set_name>:<glyph><variant_suffix>. Return None if glyph absent.

    If glyph present but suffixed glyph required & missing -> raise IconResolutionError.
    Reuses the SVG-rendering logic by inlining the same shape used by DefaultIconBackend
    (we cannot call backend.render() because that goes through the semantic mapping).
    """
    data = load_iconify_data(set_name)
    icons = data["icons"]
    if glyph not in icons:
        return None
    config = ICON_SETS[set_name]
    suffix = config.variants.get(variant)
    lookup = glyph + suffix if suffix else glyph
    if lookup not in icons:
        # Save-time validation should have caught this. Loud, not silent.
        raise IconResolutionError(
            f"Glyph {glyph!r} variant {variant!r} expected suffixed name {lookup!r} "
            f"in icon set {set_name!r}, but it is missing."
        )
    # Inline the SVG-build using the same shape as DefaultIconBackend.render
    ...


def render_course_icon(
    icon: str,
    icon_fallback: str = "",
    *,
    variant: str = "outline",
    css_class: str = "size-5",
    aria_label: str = "",
) -> SafeString:
    backend = get_icon_backend()
    # 1. empty -> default semantic
    if not icon:
        return SafeString(backend.render("course", variant=variant, css_class=css_class, aria_label=aria_label))
    # 2. semantic name
    if icon in SEMANTIC_ICON_NAMES:
        return SafeString(backend.render(icon, variant=variant, css_class=css_class, aria_label=aria_label or icon))
    # 3. literal glyph in the active set
    active = _active_set_name()
    rendered = _render_literal(active, icon, variant, css_class, aria_label or icon)
    if rendered is not None:
        return rendered
    # 4. explicit <iconset>:<glyph> fallback
    if icon_fallback and _FALLBACK_RE.match(icon_fallback):
        set_name, glyph = icon_fallback.split(":", 1)
        if set_name in ICON_SETS:
            rendered = _render_literal(set_name, glyph, variant, css_class, aria_label or glyph)
            if rendered is not None:
                return rendered
    # 5. graceful default
    return SafeString(backend.render("course", variant=variant, css_class=css_class, aria_label=aria_label or "course"))
```

**Important**: `_render_literal` must produce the same SVG shape (`<svg ... role="img" aria-label="...">{body}</svg>`) and apply the same `_validate_svg_body` check as `DefaultIconBackend`. Rather than copy/paste, refactor `DefaultIconBackend.render` to expose a small private `_build_svg(set_name, lookup_name, css_class, aria_label) -> str` helper that both code paths reuse. The public `render(...)` signature on `IconBackend` is unchanged. **This is the only icons-app edit required**; it stays internal to that app.

### `icon_validation.py` rules (pseudocode)

```python
# freedom_ls/content_engine/icon_validation.py
from django.core.exceptions import ValidationError
from freedom_ls.icons.semantic_names import SEMANTIC_ICON_NAMES
from freedom_ls.icons.mappings import ICON_SETS
from freedom_ls.icons.loader import load_iconify_data

_FALLBACK_RE = re.compile(r"^[a-z0-9_-]+:[a-z0-9_-]+$")

def validate_course_icon_fields(icon: str, icon_fallback: str) -> None:
    icon = (icon or "").strip()
    icon_fallback = (icon_fallback or "").strip()

    # Empty + empty: OK
    if not icon and not icon_fallback:
        return
    # Empty + set: error
    if not icon and icon_fallback:
        raise ValidationError({"icon_fallback": "icon_fallback set without icon..."})

    # icon set: semantic OR must be a literal glyph in at least one registered set
    if icon in SEMANTIC_ICON_NAMES:
        # icon_fallback (if any) is allowed and unused. Still validate its shape & target if provided.
        if icon_fallback:
            _validate_fallback_shape_and_target(icon_fallback)
        return

    # literal glyph path
    sets_with_glyph = []
    variant_problems: list[str] = []
    for set_name, config in ICON_SETS.items():
        try:
            data = load_iconify_data(set_name)
        except (ValueError, FileNotFoundError):
            continue
        icons = data.get("icons", {})
        if icon not in icons:
            continue
        sets_with_glyph.append(set_name)
        # Variant pairing: every non-None suffix must be present
        for variant_name, suffix in config.variants.items():
            if suffix is None:
                continue
            if (icon + suffix) not in icons:
                variant_problems.append(f"{set_name}:{variant_name}")

    if not sets_with_glyph:
        raise ValidationError({"icon": f"icon {icon!r} is not a semantic name and not present in any registered icon set ..."})

    if variant_problems:
        raise ValidationError({"icon": f"icon {icon!r} is missing variant suffix glyph(s): {variant_problems}"})

    if icon_fallback:
        _validate_fallback_shape_and_target(icon_fallback)


def _validate_fallback_shape_and_target(icon_fallback: str) -> None:
    if not _FALLBACK_RE.match(icon_fallback):
        raise ValidationError({"icon_fallback": "must match '<iconset>:<glyph>' (lowercase letters/digits/_/-)"})
    set_name, glyph = icon_fallback.split(":", 1)
    if set_name not in ICON_SETS:
        raise ValidationError({"icon_fallback": f"unknown icon set {set_name!r}"})
    data = load_iconify_data(set_name)
    icons = data.get("icons", {})
    if glyph not in icons:
        raise ValidationError({"icon_fallback": f"glyph {glyph!r} not found in icon set {set_name!r}"})
    config = ICON_SETS[set_name]
    missing = [variant for variant, suffix in config.variants.items() if suffix is not None and (glyph + suffix) not in icons]
    if missing:
        raise ValidationError({"icon_fallback": f"glyph {glyph!r} in {set_name!r} missing variant glyphs: {missing}"})
```

### `course_accent.py`

```python
# Neutral slot keys (not role names): the resolved colours live in the
# `--fls-course-accent-<slot>` token series, keeping cards decoupled from the
# semantic UI roles. Order is fixed — reordering would remap every course.
PALETTE: tuple[str, ...] = ("1", "2", "3", "4", "5")
```

### model changes (`Course`)

```python
accent_slot = models.PositiveSmallIntegerField(default=0, editable=False)

@property
def accent_role(self) -> str:
    return PALETTE[self.accent_slot]

def save(self, *args, **kwargs):
    if self._state.adding:
        self._set_site_from_request()
        self.accent_slot = Course.objects.filter(site_id=self.site_id).count() % len(PALETTE)
    super().save(*args, **kwargs)
```

The slot is assigned once on creation by per-site creation order, so the catalogue stays evenly distributed and a course keeps its colour for life. Updates never reshuffle. The data migration backfills existing rows in slug order per site.

```python
# (get_current_courses, get_completed_courses, get_recommended_courses, get_course_index)
# Accent needs no view annotation — `course.accent_role` is a model property
# backed by the stored `accent_slot`. Templates read it directly.


def _annotate_next_up(course, user):
    """Annotate an in-progress course with next_up title and continue_url."""
    children = get_course_index(user=user, course=course)
    # Flatten: if a child has nested children (CoursePart), look at those too
    flat = []
    for c in children:
        flat.append(c)
        for nested in c.get("children", []):
            flat.append(nested)
    next_item = next(
        (c for c in flat if c["status"] == "IN_PROGRESS" and c.get("url")),
        None,
    ) or next(
        (c for c in flat if c["status"] == "READY" and c.get("url")),
        None,
    )
    if next_item:
        setattr(course, "next_up_title", next_item["title"])
        setattr(course, "next_up_url", next_item["url"])
    else:
        setattr(course, "next_up_title", "")
        setattr(course, "next_up_url", "")


@login_required

    registered_courses = get_current_courses(request.user)
    completed_courses = get_completed_courses(request.user)
    recommended_courses = get_recommended_courses(request.user)

        "registered_courses": registered_courses,
        "completed_courses": completed_courses,
        "recommended_courses": recommended_courses,

    return render(request, "student_interface/dashboard.html", context)
```

The four collection partials (`current-courses`, `recommended-courses`, `learning-history`, `all-courses`) are updated to call `{% partial dispatch-card %}` instead of the legacy `{% partial course-card %}`.

The legacy `{% partialdef course-card %}` block is **deleted** once nothing references it.

### Card silhouette (shared bones, applied per partial)

All three partials use the same outer skeleton:

```html
<article class="surface relative flex flex-col overflow-hidden ...">
    {# top-aligned accent tile: course icon centred on the per-slot gradient #}
    <div class="course-accent-{{ course.accent_role|default:'1' }} flex items-center justify-center h-28" aria-hidden="true">
        {% load course_icon_tags %}
        {% course_icon course.icon course.icon_fallback css_class="size-12" aria_label=course.title %}
    </div>

    <div class="flex-1 min-w-0 space-y-2">
        <p class="text-xs text-muted uppercase tracking-wide">{{ eyebrow }}</p>

        {# stretched link: a single <a> with a ::before overlay extending the hit zone #}
        <h3>
            <a href="{{ click_destination }}" class="before:absolute before:inset-0 before:content-['']">
                {{ course.title }}
            </a>
        </h3>

        {{ state_specific_slot }}
    </div>
</article>
```

The `::before` overlay is achieved via Tailwind utility classes on the anchor (`before:absolute before:inset-0 before:content-['']`). The outer `<article>` gets `relative` so the overlay sizes to the article. Per the spec: do **not** wrap the card in `<a>`. Test selection still works because the overlay is empty.

#### State 1 — In progress

- Eyebrow: `In progress`
- Body slot: `Next up: {{ course.next_up_title }}` rendered only when `course.next_up_title`.
- Progress bar: `<progress role="progressbar" aria-valuemin="0" aria-valuemax="100" aria-valuenow="{{ course.progress_percentage }}" aria-label="Progress for {{ course.title }}">` plus a numeric `{{ course.progress_percentage }}%` label. Styled with `class="course-progress-{{ course.accent_role|default:'1' }}"` (soft track + solid fill).
- Click destination: `course.next_up_url` (computed in `_annotate_next_up`).

#### State 2 — Not started

- Eyebrow: `Not started`
- Body slot: the title; an optional `Next up:` line (when present); and a **0% progress bar** (same `course-progress-{slot}` markup as in-progress) rendered as a visual anchor so the card aligns with in-progress cards in a mixed grid.
- Click destination is split:
  - **Desktop** (`md:` and up): the card is wrapped in a `<c-modal>` whose trigger is the card itself. Modal contents include `course_preview_content.html`. The card's stretched link href becomes the mobile preview URL — but on desktop, an Alpine click handler on the article opens the modal via `x-on:click.prevent="show"`. Use the modal's existing `<c-slot name="trigger">` pattern; the modal already wraps any trigger element so it can simply contain the article.
  - **Mobile** (< md): the modal markup is hidden via `class="hidden md:block"`; the card's anchor href is `student_interface:course_preview` (new URL: `/courses/<slug>/preview/`).
- Implementation note: the simplest way to get mobile-vs-desktop split without JS branching is to render two siblings: one wrapping `<c-modal>` shown only `md:block`, one bare card link shown only `md:hidden`. Both render the same article skeleton via a small inner partialdef. CSS hides one or the other.

#### State 3 — Complete

- Eyebrow: `Completed`
- Body slot: title plus a check-shape badge: `{% icon name="check" class="size-4" %}` inside a small rounded chip with text "Completed". Per WCAG 1.4.1 the state is conveyed by text + shape; the accent colour is decorative only.
- No progress bar.
- Click destination: `student_interface:course_finish` for the slug.

### `course_preview.html` (standalone page)

Extends `_base.html`. Loads via the new view `course_preview(request, course_slug)` that renders the same context as the modal needs and `{% include "student_interface/partials/course_preview_content.html" %}`.

### URL addition

```python
path("courses/<slug:course_slug>/preview/", views.course_preview, name="course_preview"),
```

### View addition

```python
@login_required
def course_preview(request, course_slug):
    course = get_object_or_404(Course, slug=course_slug)
    children = get_course_index(user=request.user, course=course)
    is_registered = get_is_registered(user=request.user, course=course)
    return render(request, "student_interface/course_preview.html", {
        "course": course,
        "children": children,
        "is_registered": is_registered,
    })
```

## Risks & migration safety

- The URL-name rename (`home` → `dashboard`) keeps the path at `/`, so external bookmarks and `LOGIN_REDIRECT_URL` remain valid. Internal call sites are limited (one template).
- Removing `partial_list_courses` returns 404 for `/partials/courses/`. The dashboard's HTMX trigger was the only consumer; tests are removed in the same commit.
- Adding `--color-{role}-soft` tokens is purely additive; existing themes inherit the default mix.
- Adding `course` to `SEMANTIC_ICON_NAMES` and to all four mappings is enforced by `freedom_ls.E007`; CI fails fast if any mapping is missed.
- Literal-glyph and `icon_fallback` paths are net-new; no existing callers produce them, so no compatibility risk.
- `freedom_ls.icons` does **not** import from `student_interface` — the resolver lives inside `student_interface` precisely to keep the icons app a portable, downstream-installable Django app.
