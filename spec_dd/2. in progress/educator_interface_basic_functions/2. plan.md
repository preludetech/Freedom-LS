# Implementation Plan: Educator Interface Basic Functions

This plan implements four interconnected features using TDD. Each task follows RED-GREEN-REFACTOR.

**Skills to use throughout:**
- `testing` skill for all test writing (TDD)
- `template` skill for all template creation/editing
- `frontend-styling` skill for Tailwind classes — always check `tailwind.components.css` before using raw Tailwind, and extract repeated patterns into component classes
- `alpine-js` skill for client-side interactivity (modals, tabs) — always add `x-transition` when using `x-show`
- `multi-tenant` skill when working with SiteAwareModel
- `icon-usage` skill when adding icons to buttons
- `brand-guidelines` skill for visual design decisions

**Design note:** The spec mentions a `TabbedInstanceView` subclass. This plan instead adds optional `tabs` support directly to `InstanceView` (with conditional rendering). This avoids an unnecessary class hierarchy while achieving the same result — any InstanceView can define tabs without requiring a different base class.

class FormPanelAction(PanelAction):
    form_class: type[forms.ModelForm]
    form_title: str = ""

    def get_form(self, request: HttpRequest, instance: Model | None = None) -> forms.ModelForm:
        data = request.POST if request.method == "POST" else None
        return self.form_class(data, instance=instance)

    def get_form_url(self, base_url: str) -> str:
        """URL for form submission via HTMX."""
        ...

    def handle_submit(self, request: HttpRequest, instance: Model | None = None) -> HttpResponse:
        """Process form submission. Override for custom success behavior."""
        form = self.get_form(request, instance)
        if form.is_valid():
            return self.form_valid(request, form)
        return self.form_invalid(request, form)

    def form_valid(self, request: HttpRequest, form: forms.ModelForm) -> HttpResponse:
        """Override in subclasses for success behavior."""
        raise NotImplementedError

    def form_invalid(self, request: HttpRequest, form: forms.ModelForm) -> HttpResponse:
        """Return 422 with re-rendered form."""
        html = render_to_string("partials/modal_form.html", {"form": form, ...}, request=request)
        return HttpResponse(html, status=422)

    def render(self, request: HttpRequest, panel: Panel, base_url: str) -> str:
        """Render trigger button + modal with form."""
        form = self.get_form(request)
        return render_to_string("partials/modal_form.html", {...}, request=request)
```

class CreateCohortAction(FormPanelAction):
    label = "Create Cohort"
    variant = "primary"
    form_class = CohortForm
    form_title = "Create Cohort"
    action_name = "create_cohort"
    submit_buttons = [
        {"label": "Save and add another", "variant": "outline", "name": "action", "value": "save_and_add"},
        {"label": "Save", "variant": "primary"},
    ]

    def has_permission(self, request, instance=None):
        return request.user.has_perm("freedom_ls_student_management.add_cohort")

    def form_valid(self, request, form):
        cohort = form.save()
        if request.POST.get("action") == "save_and_add":
            # Return empty form, trigger datatable reload
            html = self._render_empty_form(request)
            response = HttpResponse(html)
            response["HX-Trigger"] = "cohortCreated"
            return response
        # Redirect to new cohort detail
        url = reverse("educator_interface:interface", kwargs={"path_string": f"cohorts/{cohort.pk}"})
        response = HttpResponse(status=204)
        response["HX-Redirect"] = url
        return response

        """Render panels for a specific tab (used by lazy-load HTMX calls)."""

class DeleteAction(PanelAction):
    label = "Delete"
    variant = "danger"
    action_name = "delete"

    def __init__(self, success_url: str = ""):
        self.success_url = success_url

    def get_cascade_summary(self, instance: Model) -> list[str]:
        """Use Django's Collector to show what will be cascade-deleted."""
        from django.db.models.deletion import Collector
        collector = Collector(using=instance._state.db)
        collector.collect([instance])
        summary = []
        for model, objs in collector.data.items():
            if model is not type(instance):
                count = len(objs)
                if count:
                    summary.append(f"{count} {model._meta.verbose_name_plural}")
        return summary

    def render(self, request, panel, base_url):
        cascade = self.get_cascade_summary(panel.instance)
        return render_to_string(
            "partials/delete_confirmation.html",
            {"instance": panel.instance, "cascade_summary": cascade,
             "delete_url": f"{base_url}/__actions/delete", "variant": self.variant},
            request=request,
        )

    def handle_submit(self, request, instance):
        instance.delete()
        response = HttpResponse(status=204)
        response["HX-Redirect"] = self.success_url
        return response

    def has_permission(self, request, instance=None):
        if instance is None:
            return False
        model_name = instance._meta.model_name
        app_label = instance._meta.app_label
        return request.user.has_perm(f"{app_label}.delete_{model_name}", instance)
```
